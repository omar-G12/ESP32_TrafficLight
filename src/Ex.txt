void trafficTask(void *parameters)
{
    ledstuffstruct *data = (ledstuffstruct *)parameters;
    gpio_num_t greenpin = data->_Green_pin;
    gpio_num_t yellowpin = data->_Yellow_pin;
    gpio_num_t Redpin = data->_Red_pin;
    int Green_delay = data->green_blink;
    int Yellow_delay = data -> yellow_blink;
    int Red_delay = data -> red_blink;

    while (1) {
        switch(current){
            case Green: 
             gpio_set_level(greenpin, 1);
             vTaskDelay(Green_delay / portTICK_PERIOD_MS);
             gpio_set_level(greenpin, 0);
             vTaskDelay(Green_delay / portTICK_PERIOD_MS);
             break;
            case Yellow: 
             gpio_set_level(yellowpin, 1);
             vTaskDelay(Yellow_delay / portTICK_PERIOD_MS);
             gpio_set_level(yellowpin, 0);
             vTaskDelay(Yellow_delay / portTICK_PERIOD_MS);
             break;
            case Red: 
             gpio_set_level(Redpin, 1);
             vTaskDelay(Red_delay / portTICK_PERIOD_MS);
             gpio_set_level(Redpin, 0);
             vTaskDelay(Red_delay / portTICK_PERIOD_MS);
             break;
        }
    }
}

void button_task(void *arg)
{
    bool led_status = false; 
    while (1){
        vTaskSuspend(NULL);
        led_status = true; 
        gpio_set_level(CONFIG_LED_PIN,led_status);
        printf("Button pressed! \n");
    }
}
